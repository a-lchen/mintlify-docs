每个 Mintlify 网站都需要一个`docs.json`文件的核心配置设置。了解更多关于[属性](#properties)如下。

## 属性

<ResponseField name="name" type="string" required>
  项目名称。用于全局标题。

  例子：`mintlify`
</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  一个包含该组内所有页面的组数组

  <Expandable title="Navigation">
    <ResponseField name="group" type="string">
      该组的名称。

      示例：`Settings`
    </ResponseField>

    <ResponseField name="pages" type="string[]">
      作为页面的 markdown 文件的相对路径。

      示例：`["customization", "page"]`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  徽标图像的路径或包含“明亮”和“黑暗”模式徽标图像路径的对象

  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      浅色模式下 logo 的路径
    </ResponseField>

    <ResponseField name="dark" type="string">
      深色模式下的标志路径
    </ResponseField>

    <ResponseField name="href" type="string" default="/">
      点击徽标会链接到的位置
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  favicon 图像的路径
</ResponseField>

<ResponseField name="colors" type="Colors">
  全局主题的十六进制颜色代码

  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>
      主要颜色。通常用于高亮内容、部分标题、强调色，在浅色模式中
    </ResponseField>

    <ResponseField name="light" type="string">
      暗模式的主色。常用于高亮内容、部分标题、强调色，在暗模式下
    </ResponseField>

    <ResponseField name="dark" type="string">
      重要按钮的主要颜色
    </ResponseField>

    <ResponseField name="background" type="object">
      浅色和深色模式下的背景颜色

      <Expandable title="Object">
        <ResponseField name="light" type="string" required>
          浅色模式下背景的十六进制颜色代码
        </ResponseField>

        <ResponseField name="dark" type="string" required>
          暗模式下背景的十六进制颜色代码
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  数组`name`和`url`您想在顶部栏中包含的链接列表

  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
      按钮的名称。

      示例：
      `Contact us`
    </ResponseField>

    <ResponseField name="url" type="string">
      单击按钮后的 URL。 例如：`https://mintlify.com/docs`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="Topbar Call to Action">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
      Link 显示一个按钮。GitHub 显示在提供的 url 中的仓库信息，包括 GitHub 星标的数量。
    </ResponseField>

    <ResponseField name="url" type="string">
      如果`link`：按钮链接到哪里。

      如果`github`: 链接到存储库以加载 GitHub 信息。
    </ResponseField>

    <ResponseField name="name" type="string">
      按钮内的文字。仅在必要时`type` 是一个 `link`。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅在您希望通过导航栏中的下拉菜单显示不同版本的文档时使用此功能。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  锚点的数组， 包括`icon`，`color`，和`url`.

  <Expandable title="Anchor">
    <ResponseField name="icon" type="string">
      这个[Font Awesome](https://fontawesome.com/search?q=heart)用于突出显示锚点的图标。

      示例：`comments`
    </ResponseField>

    <ResponseField name="name" type="string">
      锚点标签的名称。

      示例：
      `Community`
    </ResponseField>

    <ResponseField name="url" type="string">
      标记放入锚点页面的URL起始部分。通常，这就是您将页面放入其中的文件夹的名称。
    </ResponseField>

    <ResponseField name="color" type="string">
      锚点图标背景的十六进制颜色。如果传递具有属性的对象，也可以是渐变色`from`和`to` 它们每个都是一个十六进制颜色。
    </ResponseField>

    <ResponseField name="version" type="string">
      如果您想隐藏锚点直到选择正确的文档版本，可使用此选项。
    </ResponseField>

    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      传递`true`如果您想隐藏锚点，直到您直接链接某人到其中的文档。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      以下之一: "brands"、"duotone"、"light"、"sharp-solid"、"solid" 或 "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖最顶层锚点的默认配置。

  <Expandable title="Object">
    <ResponseField name="name" default="Documentation" type="string">
      最顶层锚点的名称
    </ResponseField>

    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome 图标。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      其中之一: "brands", "duotone", "light", "sharp-solid", "solid", 或 "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  导航标签的数组。

  <Expandable title="Tabs">
    <ResponseField name="name" type="string">
      选项卡标签的名称。
    </ResponseField>

    <ResponseField name="url" type="string">
      URL 的起始部分，标记哪些页面进入该选项卡。通常，这是您放置页面的文件夹名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  API 设置配置。了解更多关于 API 页面的信息，请访问[API 组件](/api-playground/demo)。

  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      所有API端点的基本URL。如果`baseUrl` 是一个数组，它将为用户提供可以切换的多个基础 URL 选项。
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="Auth">
        <ResponseField name="method" type="&#x22;bearer&#x22; | &#x22;basic&#x22; | &#x22;key&#x22;">
          用于所有API端点的认证策略。
        </ResponseField>

        <ResponseField name="name" type="string">
          API 使用中的身份验证参数名称。

          如果方法是`basic`，格式应该是`[usernameName]:[passwordName]`
        </ResponseField>

        <ResponseField name="inputPrefix" type="string">
          默认值被设计为身份验证输入字段的前缀。

          例如，如果一个`inputPrefix` 的 `AuthKey`会继承认证字段的默认输入结果为`AuthKey`.
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API 演练场的配置

      <Expandable title="Playground">
        <ResponseField name="mode" default="show" type="&#x22;show&#x22; | &#x22;simple&#x22; | &#x22;hide&#x22;">
          播放区是否显示、隐藏，或仅显示端点而无用户交互`simple`

          了解更多[操场指南](/api-playground/demo)
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="maintainOrder" type="boolean">
      启用此标志可确保 OpenAPI 页面中的键顺序与 OpenAPI 文件中定义的键顺序相匹配。

      <Warning>此行为即将默认启用，届时该字段将被弃用。</Warning>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  指向您的 OpenAPI 文件的 URL 字符串或字符串数组或相对路径。

  示例：

  <CodeGroup>
    ```json Absolute
    "openapi": "https://example.com/openapi.json"
    ```

    ```json Relative
    "openapi": "/openapi.json"
    ```

    ```json Multiple
    "openapi": ["https://example.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
    ```
  </CodeGroup>
</ResponseField>

<ResponseField name="footerSocials" type="FooterSocials">
  一个社交媒体账户的对象，其中键值对代表社交媒体平台和账户网址。

  示例：

  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```

  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
      以下值之一`website`，`facebook`,&#x20;
      `x`，`discord`，`slack`，`github`，`linkedin`，`instagram`, `hacker-news`

      示例：
      `x`
    </ResponseField>

    <ResponseField name="property" type="string">
      社交平台的 URL。

      示例：`https://x.com/mintlify`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback" type="Feedback">
  启用反馈按钮的配置

  <Expandable title="Feedback">
    <ResponseField name="suggestEdit" type="boolean" default="false">
      启用一个按钮，以允许用户通过 pull requests 建议编辑
    </ResponseField>

    <ResponseField name="raiseIssue" type="boolean" default="false">
      启用一个按钮，允许用户对文档提出问题
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="modeToggle" type="ModeToggle">
  自定义暗模式切换。

  <Expandable title="ModeToggle">
    <ResponseField name="default" type={'"light" or "dark"'}>
      设置是否始终为新用户显示亮色或暗色模式。如果未设置，我们默认使用与用户操作系统相同的模式。
    </ResponseField>

    <ResponseField name="isHidden" type="boolean" default="false">
      设置为 true 以隐藏深色/浅色模式切换。你可以结合`isHidden` 与 `default`以强制您的文档仅使用亮模式或暗模式。例如：

      <CodeGroup>
        ```json Only Dark Mode
        "modeToggle": {
          "default": "dark",
          "isHidden": true
        }
        ```

        ```json Only Light Mode
        "modeToggle": {
          "default": "light",
          "isHidden": true
        }
        ```
      </CodeGroup>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  每个页面背后显示的背景图片。请参阅示例与[Infisical](https://infisical.com/docs) 和 [FRPC](https://frpc.io)。
</ResponseField>
