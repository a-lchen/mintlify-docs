每个 Mintlify 页面都需要一个`docs.json`包含核心配置设置的文件。了解更多关于[特性](#properties) 下方。

## 属性

<ResponseField name="name" type="string" required>
  您项目的名称。用于全局标题中。

  例子：`mintlify`
</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  带有此组中所有页面的组数组

  <Expandable title="Navigation">
    <ResponseField name="group" type="string">
      该组的名称。

      示例：`Settings`
    </ResponseField>

    <ResponseField name="pages" type="string[]">
      Markdown 文件的相对路径，将用作页面。

      示例：`["customization", "page"]`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  Logo 图片的路径，或一个包含 "light" 和 "dark" 模式下 Logo 图片路径的对象

  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      Light 模式下的 Logo 路径
    </ResponseField>

    <ResponseField name="dark" type="string">
      暗模式下的徽标路径
    </ResponseField>

    <ResponseField name="href" type="string" default="/">
      点击徽标后的跳转位置
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  Favicon 图像的路径
</ResponseField>

<ResponseField name="colors" type="Colors">
  全局主题的十六进制颜色代码

  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>
      主要颜色。在浅色模式下，最常用于突出显示的内容、章节标题和重点
    </ResponseField>

    <ResponseField name="light" type="string">
      深色模式的主色。最常用于深色模式下的突出内容、段落标题和强调色
    </ResponseField>

    <ResponseField name="dark" type="string">
      重要按钮的主色
    </ResponseField>

    <ResponseField name="background" type="object">
      在亮模式和暗模式下背景的颜色

      <Expandable title="Object">
        <ResponseField name="light" type="string" required>
          Light 模式下背景的十六进制颜色代码
        </ResponseField>

        <ResponseField name="dark" type="string" required>
          暗模式下背景的十六进制颜色代码
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  数组 von`name`s 和`url`在顶部栏中嵌入的链接的来源

  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
      按钮的名称。

      示例：`Contact us`
    </ResponseField>

    <ResponseField name="url" type="string">
      单击按钮时的 URL。示例：`https://mintlify.com/docs`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="Topbar Call to Action">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
      链接显示一个按钮。GitHub 在指定的 URL 下显示仓库信息，包括 GitHub 星数。
    </ResponseField>

    <ResponseField name="url" type="string">
      Wenn`link`: 按钮链接的位置。

      如果`github`：链接到要从中加载 GitHub 信息的存储库。
    </ResponseField>

    <ResponseField name="name" type="string">
      按钮内的文本。仅在
      `type`一个`link`ist。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅当您希望在导航栏中使用下拉列表显示文档的不同版本时，才使用此功能。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  一个包含锚点的数组`icon`，`color`，和`url`.

  <Expandable title="Anchor">
    <ResponseField name="icon" type="string">
      Das[Font Awesome](https://fontawesome.com/search?q=heart)-用于锚点的符号。

      示例：`comments`
    </ResponseField>

    <ResponseField name="name" type="string">
      锚点标签的名称。

      例如：`Community`
    </ResponseField>

    <ResponseField name="url" type="string">
      表示属于该锚点的页面的URL开始部分。通常这是您存放页面的文件夹名称。
    </ResponseField>

    <ResponseField name="color" type="string">
      锚点符号背景的十六进制颜色。 如果使用带有属性的对象，它也可以是渐变色。`from`和`to`传递的值分别是十六进制颜色。
    </ResponseField>

    <ResponseField name="version" type="string">
      在选择正确的文档版本之前，如果您想隐藏一个锚点，可以使用。
    </ResponseField>

    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      传递给`true`，如果您想要隐藏锚点，直到您直接将某人链接到其中的文档。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      一个是："brands", "duotone", "light", "sharp-solid", "solid" 或 "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖顶层锚点的默认配置。

  <Expandable title="Object">
    <ResponseField name="name" default="Documentation" type="string">
      最上方锚点的名称
    </ResponseField>

    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome-符号。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      其中之一："brands"， "duotone"， "light"， "sharp-solid"， "solid" 或 "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  一个导航选项卡的数组。

  <Expandable title="Tabs">
    <ResponseField name="name" type="string">
      标签名称。
    </ResponseField>

    <ResponseField name="url" type="string">
      URL 的开头，用于标记哪些页面属于该选项卡。通常，这是您存放页面的文件夹的名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  API设置的配置。了解有关API页面的更多信息[API组件](/api-playground/demo)。

  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      所有 API 端点的基本 URL。如果`baseUrl`是一个数组，允许多个基础URL选项，用户可以在其中选择。
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="Auth">
        <ResponseField name="method" type="&#x22;bearer&#x22; | &#x22;basic&#x22; | &#x22;key&#x22;">
          用于所有API端点的认证策略。
        </ResponseField>

        <ResponseField name="name" type="string">
          在 API 操作平台使用的身份验证参数的名称。

          如果方法`basic` 是，应使用的格式 `[usernameName]:[passwordName]`
        </ResponseField>

        <ResponseField name="inputPrefix" type="string">
          作为身份验证输入字段前缀的默认值。

          例如，如果一个`inputPrefix` von `AuthKey`将继承认证字段的默认输入结果为`AuthKey`。
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API 操作台的配置

      <Expandable title="Playground">
        <ResponseField name="mode" default="show" type="&#x22;show&#x22; | &#x22;simple&#x22; | &#x22;hide&#x22;">
          Playground 是显示、隐藏或仅显示端点而无额外用户交互的`simple`

          了解更多信息，请访问[Playground 指南](/api-playground/demo)
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="maintainOrder" type="boolean">
      启用此标志可确保 OpenAPI 页面中的键顺序与 OpenAPI 文件中定义的顺序一致。

      <Warning>这种行为很快将被默认启用，届时该字段将被标记为废弃。</Warning>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  一个字符串或一个字符串数组，包含指向您的 OpenAPI 文件的 URL 或相对路径。

  示例：

  <CodeGroup>
    ```json Absolute
    "openapi": "https://example.com/openapi.json"
    ```

    ```json Relative
    "openapi": "/openapi.json"
    ```

    ```json Multiple
    "openapi": ["https://example.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
    ```
  </CodeGroup>
</ResponseField>

<ResponseField name="footerSocials" type="FooterSocials">
  一个社交媒体帐户对象，其中键:属性对表示社交媒体平台和帐户URL。

  例子：

  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```

  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
      以下其中一个值`website`，`facebook`, `x`, `discord`，`slack`，
      `github`，
      `linkedin`，`instagram`，`hacker-news`

      例子：`x`
    </ResponseField>

    <ResponseField name="property" type="string">
      社交媒体平台的 URL。

      例子：`https://x.com/mintlify`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback" type="Feedback">
  用于启用反馈按钮的配置

  <Expandable title="Feedback">
    <ResponseField name="suggestEdit" type="boolean" default="false">
      启用一个按钮，允许用户通过拉取请求提交更改建议
    </ResponseField>

    <ResponseField name="raiseIssue" type="boolean" default="false">
      启用一个按钮，使用户可以报告文档问题
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="modeToggle" type="ModeToggle">
  调整黑暗模式开关。

  <Expandable title="ModeToggle">
    <ResponseField name="default" type={'"light" or "dark"'}>
      设置是否要为新用户始终显示浅色模式或深色模式。如果未设置，我们将默认使用与用户操作系统相同的模式。
    </ResponseField>

    <ResponseField name="isHidden" type="boolean" default="false">
      设置为 true 以隐藏明暗模式切换开关。您可以`isHidden` mit `default`结合，强制您的文档只使用明亮或黑暗模式。例如：

      <CodeGroup>
        ```json Only Dark Mode
        "modeToggle": {
          "default": "dark",
          "isHidden": true
        }
        ```

        ```json Only Light Mode
        "modeToggle": {
          "default": "light",
          "isHidden": true
        }
        ```
      </CodeGroup>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  在每个页面后面显示的背景图像。参见示例与[Infisical](https://infisical.com/docs) 和 [FRPC](https://frpc.io)。
</ResponseField>
